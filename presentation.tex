\begin{frame}{SLAM: Mapping the Environment}
  \begin{columns}[T]
    \begin{column}{0.55\textwidth}
      \textbf{Pipeline}
      \begin{itemize}
        \item \texttt{slam\_toolbox} (online async mode) builds \texttt{/map} from \texttt{/scan}
        \item \texttt{explore\_lite} drives the robot to frontiers autonomously
        \item EKF (\texttt{robot\_localization}) fuses wheel odometry + IMU
        \item Map saved as PGM/YAML on completion (\textasciitilde 956$\times$1270 cells, 0.05 m resolution)
      \end{itemize}
      \vspace{0.4em}
      \textbf{QR landmark detection (parallel)}
      \begin{itemize}
        \item \texttt{qr\_localizer} runs on all 4 camera streams
        \item Bearing-from-camera + range-from-lidar fusion
        \item Quality-weighted running mean per label
        \item Output: \texttt{landmarks.yaml}
      \end{itemize}
    \end{column}

\begin{frame}{Dynamic Obstacle Avoidance \& Replanning}
  \begin{columns}[T]
    \begin{column}{0.55\textwidth}
      \textbf{Costmap layers (Nav2)}
      \begin{itemize}
        \item \texttt{static\_layer}: saved map (walls)
        \item \texttt{obstacle\_layer}: live \texttt{/scan} (new obstacles)
        \item \textbf{Dual inflation:}
          walls 1.5 m halo, \\ dynamic obstacles 0.3 m halo
      \end{itemize}
      \vspace{0.4em}
      \textbf{Local controller: MPPI}
      \begin{itemize}
        \item 1200 sampled trajectories per cycle, 2 s horizon
        \item Critics: CostCritic, PathAlign, PathFollow, \dots
        \item Auto-disables PathAlign when $>$\,7\% of path blocked
      \end{itemize}
      \vspace{0.4em}
      \textbf{Replanning}
      \begin{itemize}
        \item BT triggers \texttt{ComputePathToPose} at \textasciitilde 1 Hz
        \item New obstacle in global costmap $\Rightarrow$ path re-routes
        \item Collision monitor as safety fallback
      \end{itemize}
